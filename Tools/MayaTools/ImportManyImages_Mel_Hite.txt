proc ImportMultiplePhotos(string $filePath, int $startingPhoto, int $numOfPhotos)
{
    int $input = $numOfPhotos;
    for ($i = 0; $i < $input; $i++)
    {
        imagePlane -fn ($filePath + " (" + ($startingPhoto + $i) + ")" + ".jpg") ;
    }
}
